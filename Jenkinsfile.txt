pipeline {
    agent any
    stages {
        stage("Checkout"){
            steps {
                checkout scmGit(branches: [[name: '*/master']], extensions: [], userRemoteConfigs: [[url: 	'https://github.com/Venkateshgithubb/ustreamer']])
            }
        }
        stage("Verify"){
            steps {
                sh "make --version"
            }
        }
        stage("Build"){
            steps {
                 sh "make"
            }
        }
 }
}
